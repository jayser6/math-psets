\section{}
Identify (up to isomorphism) all finite groups $G$ which have exactly three irreducible
characters of dimensions $1$, $2$, and $3$, respectively.
(You may use the \textit{Sylow theorems}, which you can look up if they weren't covered
in your 100A.)
 
\section{}
Let $G$ be a finite cyclic group. Let $S$ be the set of generators of $G$. Let $\chi$ be
any character of $G$ (not necessarily irreducible).
 
\begin{enumerate}
    \item Show that
    \[
        \sum_{g \in S} \chi(g)
    \]
    is an integer. (\textit{Hint}: reduce to the irreducible case.)
 
    \item Show that
    \[
        \prod_{g \in S} |\chi(g)|^{2}
    \]
    is a \emph{nonnegative} integer.
 
    \item Show that if $\chi(g) \neq 0$ for all $g \in S$,
    \[
        \sum_{g \in S} |\chi(g)|^{2} \geq |S|.
    \]
    (\textit{Hint}: combine the previous part with the arithmetic-geometric mean
    inequality.)
\end{enumerate}
 
\section{}
Let $G$ be a finite group. Let $\chi$ be an irreducible character such that
$\chi(g) \neq 0$ for all $g \in S$. Use the previous exercise to deduce that
\[
    \sum_{g \in G \setminus \{1\}} |\chi(g)|^{2} \geq |G| - 1
\]
and deduce that $\chi(1) = 1$. In other words, an irreducible character $\chi$ of
dimension greater than $1$ must take the value $0$ somewhere; this theorem is originally
due to Burnside.
 
\section{}
Let $G$ be a finite group. Prove that the following two conditions are equivalent.
\begin{enumerate}
    \item The entries of the character table of $G$ are all integers.
    \item Any two elements of $G$ which generate the same cyclic subgroup of $G$ belong
    to the same conjugacy class.
\end{enumerate}
(\textit{Hint}: use orthogonality to show that any two elements of $G$ which have the
same value on every character are conjugate.)
Then deduce as a corollary that for every positive integer $n$, the character table of
$S_n$ contains only integers.
